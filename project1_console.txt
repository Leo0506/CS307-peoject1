
create table students (
    student_id int primary key ,
    name varchar(40) not null,
    gender int,
    college_id int references colleges(college_id)

);


create table colleges (
    college_id serial primary key ,
    chinese_name varchar(20) not null ,
    english_name varchar(20) not null ,
    unique (chinese_name),
    unique (english_name)
);

create table departments (
    department_id serial primary key ,
    department_name varchar(40) not null,
    unique (department_name)
);


create table prerequisites__courses(
    pc_id serial primary key ,
    course_id int references courses(course_id) not null ,
    prerequisites bytea
);


create table classes (
    class_id serial primary key ,
    class_name varchar(80) not null,
    total_capacity int,
    course_id int references courses(course_id),
    department_id int references departments(department_id)
);


create table courses(
    course_id serial primary key,
    course_code varchar(20) not null,
    course_name varchar(80) not null,
    course_hour int,
    course_credit int not null,
    unique (course_code)
);


create table students__courses (
    sc_id serial primary key ,
    student_id int references students(student_id) not null ,
    course_id int references courses(course_id) not null
);

create table teachers (
    teacher_id serial primary key ,
    teacher_name varchar(40) not null
);


create table classes__teachers (
    ct_id serial primary key ,
    teacher_id int references teachers(teacher_id) not null ,
    class_id int references classes(class_id) not null
);

create table weeklists (
        weeklist_id serial primary key ,
        weeks boolean[] not null,
        unique (weeks)

);

create table classlists (
    classlist_id serial primary key ,
    weeklist_id int references weeklists(weeklist_id) not null ,
    class_id int references classes(class_id) not null ,
    location varchar(80),
    weekday int not null ,
    begin_time int not null ,
    end_time int not null
);
